Promise 限制并发：写一个类 MyQueue，包含 add 方法，给后台发送请求（调用 fetch 即可），限制同一时刻最多 3 个请求。（不考虑发送失败的情况）

class MyQueue {
  constructor(max) {
    this.max = max;
    this.doingCount = 0;
    this.unDoList = [];
  }
  add(url) {
    return new Promise((resolve, reject) => {
      this.unDoList.push({ url, resolve });
      this.start();
    });
  }

  start() {
    if (this.doingCount < this.max && this.unDoList.length) {
      const item = this.unDoList.shift();
      if (item) {
        this.doingCount++;
        fetch(item.url).then((data) => {
          this.doingCount--;
          this.start();
          item.resolve(data);
        });
      }
    }
  }
}
